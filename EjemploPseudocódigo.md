## 🗺️ Algoritmo:

Algoritmo sumaDeDosNumeros

	//Variables
	Definir  a Como Real; 
	Definir b Como Real;
	Definir resultado Como Real;
	
	//Datos de entrada
	Escribir "Ingrese valor de a";
	Leer a;
	
	Escribir "Ingrese el valor de b";
	Leer b;
	
	//Datos de salida
	resultado = a + b;
	Escribir "El resultado es: " , resultado;
	
	
FinAlgoritmo
